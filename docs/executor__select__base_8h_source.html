259 template <typename ImplT, typename SysWrapperT>
260 void ExecutorSelectBase<ImplT, SysWrapperT>::iterateResult(const int aResult)
261 {
262  std::size_t processed = 0;
263  const std::size_t readyFDCount = static_cast<std::size_t>(aResult);
264  for (std::size_t i = 0, processedCount = 0; processedCount < readyFDCount;
265  ++i)
266  {
267  processed = 0;
268  if (FD_ISSET(impl().nativeHandleImpl(i), &readfds_))
269  {
270  // data can be read from socket without blocking
271  HandlerSelectBase<SysWrapperT> *handler = fdInfos_[i]->handler_;
272  handler->inDataHandler_(*fdInfos_[i], handler);
273  processed = 1;
274  }
275  if (FD_ISSET(impl().nativeHandleImpl(i), &writefds_))
276  {
277  // data can be written to socket without blocking
278  HandlerSelectBase<SysWrapperT> *handler = fdInfos_[i]->handler_;
279  handler->outDataHandler_(*fdInfos_[i], handler);
280  processed = 1;
281  }
282  if (FD_ISSET(impl().nativeHandleImpl(i), &exceptfds_))
283  {
284  // exception conditions occured in this socket can be handled
285  // without blocking
286  HandlerSelectBase<SysWrapperT> *handler = fdInfos_[i]->handler_;
287  handler->exceptCondHandler_(*fdInfos_[i], handler);
288  processed = 1;
289  }
290  processedCount += processed;
